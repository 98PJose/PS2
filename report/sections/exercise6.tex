% =====================================================================
\section{Exercise 6 -- Lo (2007) stop-loss policy: simulation check of
equation (26)}\label{sec:ex6}
% =====================================================================

We verify equation~(26) of Lo (2007, Section~4.3) by simulation. The
risky asset follows a stationary AR(1) process; a stop-loss rule moves the
portfolio into the risk-free asset whenever the previous-period strategy
return falls at or below a threshold $\zeta$. We reproduce Lo's
\textbf{Table~1} from the closed-form expression, then generate the same
table by Monte-Carlo simulation of equations~(23)--(25), and confirm that
the two agree to the printed precision.

% ---------------------------------------------------------------
\subsection{Theory}
% ---------------------------------------------------------------

The strategy return is an AR(1) with Gaussian innovations,
\begin{equation}\label{eq:lo23}
R_t \;=\; \mu \,+\, \rho\,(R_{t-1}-\mu) \,+\, \varepsilon_t,
\qquad \varepsilon_t \stackrel{\text{IID}}{\sim} \mathcal{N}\bigl(0,\sigma_\varepsilon^2\bigr),
\end{equation}
with unconditional mean $\mu$ and unconditional variance
$\sigma^2=\sigma_\varepsilon^2/(1-\rho^2)$. The stop-loss switch
\begin{equation}\label{eq:lo24}
\omega_t \;=\;
\begin{cases}
1 & \text{if } R_{t-1}>\zeta,\\
0 & \text{if } R_{t-1}\le\zeta,
\end{cases}
\end{equation}
defines the combined portfolio return
\begin{equation}\label{eq:lo25}
R_{pt} \;=\; \omega_t\,R_t \;+\; (1-\omega_t)\,R_f.
\end{equation}
Lo's AP decomposition (his eq.~22a) reads, for any policy that depends
only on past information,
\[
\E[R_{pt}] \;=\; \Cov[\omega_t,R_t] \;+\; \E[\omega_t]\,\E[R_t]
                 \;+\; (1-\E[\omega_t])\,R_f.
\]

\paragraph{Closed form.} Under \eqref{eq:lo23}--\eqref{eq:lo25} the
unconditional law of $R_{t-1}$ is $\mathcal{N}(\mu,\sigma^2)$ and the
conditional mean of the strategy return is
$\E[R_t\given R_{t-1}]=\mu+\rho(R_{t-1}-\mu)$. Letting $z\equiv(\zeta-\mu)/\sigma$,
\[
\E[\omega_t]=1-\Phi(z),\qquad
\E[(R_{t-1}-\mu)\,\omega_t]
   = \sigma\int_{z}^{\infty}\!u\,\phi(u)\,du
   = \sigma\,\phi(z),
\]
so $\Cov[\omega_t,R_t]=\rho\,\E[(R_{t-1}-\mu)\,\omega_t]=\rho\sigma\,\phi(z)$.
Substituting yields Lo's equation~(26):
\begin{equation}\label{eq:lo26}
\boxed{\;
\E[R_{pt}] \;=\;
   \underbrace{\rho\,\sigma\,\phi\!\left(\tfrac{\zeta-\mu}{\sigma}\right)}_{\text{active}}
   \;+\;
   \underbrace{\mu\bigl(1-\Phi(\tfrac{\zeta-\mu}{\sigma})\bigr)
              + R_f\,\Phi(\tfrac{\zeta-\mu}{\sigma})}_{\text{passive}}.
\;}
\end{equation}
The sign of the active component coincides with the sign of $\rho$: with
momentum ($\rho>0$) the stop-loss adds value (a low past return signals a
low future return, so switching to the risk-free is profitable); with
mean reversion ($\rho<0$) the policy is counterproductive because the
post-threshold rebound is exactly what the strategy gives up.

% ---------------------------------------------------------------
\subsection{Simulation design}
% ---------------------------------------------------------------

The benchmark parameters are those used in Lo's Table~1. Annualised
inputs are $R_f=\ExSixRfPct\%$, $\E[R_t]=\ExSixMuPct\%$ and
$\sqrt{\Var[R_t]}=\ExSixSigmaPct\%$; these are converted to monthly
moments $\mu=\E[R_t]/12$, $R_f^{(\text{m})}=R_f/12$ and
$\sigma=\ExSixSigmaPct\%/\sqrt{12}$. We sweep
$\rho\in\{-25\%,0\%,25\%,50\%\}$ and
$\zeta\in\{-1.0\%,-0.5\%,0.0\%,+0.5\%,+1.0\%\}$.

For each $(\rho,\zeta)$ cell we simulate $T=\ExSixNobs$ consecutive
realisations of \eqref{eq:lo23} (drawing $R_0$ from the stationary
distribution), apply the switch \eqref{eq:lo24}--\eqref{eq:lo25} and
estimate the AP decomposition by sample moments. The independent
seeds are derived from a base seed $\ExSixSeed$. All numbers reported
below are \emph{annualised} ($\times\ExSixFreq$) and expressed in
percent, exactly as in Lo's Table~1.

\subsection{Discussion}
% ---------------------------------------------------------------

\begin{enumerate}\setlength\itemsep{0pt}
\item \textbf{Sign of the active component.} The active term
$\rho\sigma\phi((\zeta-\mu)/\sigma)$ is non-negative iff $\rho\ge 0$.
With $\rho=-25\%$ the policy \emph{destroys} value at every threshold
($-6.6\%$ to $-6.9\%$ annualised). With $\rho=+50\%$ it \emph{adds}
$13.1\%$ to $13.8\%$, more than doubling the buy-and-hold return.
The simulation reproduces this sign and magnitude exactly.
\item \textbf{Effect of the threshold.} As $\zeta$ rises, $\Phi(z)$
rises and $\E[\omega]$ falls (more time spent in the risk-free), so the
passive component drifts toward $R_f$. The active component
$\rho\sigma\phi(z)$ peaks where $z=0$, i.e.\ around
$\zeta\approx\mu=\ExSixMuPct\%/12$. For $\rho=+50\%$ the peak is
near $\zeta=0$, consistent with Table~\ref{tab:6-theory}.
\item \textbf{The case $\rho=0$.} When returns are i.i.d., past returns
carry no information for the future, hence $\Cov[\omega_t,R_t]=0$ by
construction; the policy delivers a passive convex combination of
$\E[R_t]$ and $R_f$ weighted by $\E[\omega]$. Both columns in our
tables show \%Active is statistically indistinguishable from zero in
that row.
\end{enumerate}

\paragraph{Conclusion.} The Monte-Carlo experiment confirms Lo's
equation~(26) numerically across the full $(\rho,\zeta)$ grid of
Table~1. The maximum absolute deviation between simulation and theory
is $\ExSixMaxDiff\%$ annualised, attributable to ordinary $1/\sqrt{T}$
sampling noise.
